        let score = 0;
        let currentQ = 0;
        let timer = 120;
        let timerInterval;

        function startGame() {
            document.getElementById('screen-start').classList.add('hidden');
            document.getElementById('screen-game').classList.remove('hidden');
            document.getElementById('game-hud').classList.remove('hidden');
            renderQuestion();
            startTimer();
        }

        function startTimer() {
            timerInterval = setInterval(() => {
                timer--;
                document.getElementById('timer-text').innerText = timer + "s";
                if (timer <= 0) endGame();
            }, 1000);
        }

        function renderQuestion() {
            const q = QUESTIONS[currentQ];
            const descEl = document.getElementById('level-desc');
            descEl.innerText = q.desc;
            descEl.className = `text-3xl md:text-4xl font-bold mb-12 leading-tight ${q.color}`;
            
            const grid = document.getElementById('choices-grid');
            grid.innerHTML = '';
            
            const shuffledChoices = [...q.choices].sort(() => Math.random() - 0.5);
            
            shuffledChoices.forEach((choice, idx) => {
                const btn = document.createElement('button');
                btn.className = `choice-card w-full p-6 rounded-[2rem] text-xl font-bold text-left flex items-center gap-4 bg-blue-500 text-white shadow-[0_6px_0_#2563EB]`;
                btn.innerHTML = `<span class="bg-white/20 w-12 h-12 rounded-full flex items-center justify-center text-xl font-bold shrink-0">${idx === 0 ? 'A' : 'B'}</span> <span class="normal-weight">${choice.t}.</span>`;
                btn.onclick = () => selectChoice(choice);
                grid.appendChild(btn);
            });

            const progress = ((currentQ) / QUESTIONS.length) * 100;
            document.getElementById('progress-line').style.width = progress + "%";
        }

        function selectChoice(choice) {
            score += choice.pts;
            document.getElementById('score-text').innerText = score;

            const modal = document.getElementById('modal-feedback');
            const title = document.getElementById('fb-title');
            const emoji = document.getElementById('fb-emoji');
            
            if (choice.pts > 0) {
                title.innerText = "MUITO BEM!";
                title.className = "text-3xl font-bold mb-2 text-green-600";
                emoji.innerText = "🌟";
            } else {
                title.innerText = "ATENÇÃO!";
                title.className = "text-3xl font-bold mb-2 text-orange-500";
                emoji.innerText = "💡";
            }

            document.getElementById('fb-text').innerHTML = choice.fb;
            modal.classList.remove('hidden');
        }

        function closeFeedback() {
            document.getElementById('modal-feedback').classList.add('hidden');
            currentQ++;
            if (currentQ < QUESTIONS.length) {
                renderQuestion();
            } else {
                endGame();
            }
        }

        function endGame() {
            clearInterval(timerInterval);
            document.getElementById('screen-game').classList.add('hidden');
            document.getElementById('game-hud').classList.add('hidden');
            document.getElementById('screen-end').classList.remove('hidden');
            
            document.getElementById('final-score').innerText = score;
            const msg = document.getElementById('final-message');
            
            if (score === 1000) msg.innerHTML = "Excelente! Você é um verdadeiro mestre e sabe exatamente como cuidar dos recursos da sua família!";
            else if (score >= 700) msg.innerHTML = "Bom trabalho! Suas escolhas ajudaram muito, mas tente observar ainda mais os detalhes para não desperdiçar nada.";
            else msg.innerHTML = "O desperdício ganhou desta vez. Que tal pensar mais em cada situação e tentar de novo para melhorar sua pontuação?";
        }
